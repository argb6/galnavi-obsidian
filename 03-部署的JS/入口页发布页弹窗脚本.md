# 入口页发布页弹窗脚本

> ！位置
> 
> 入口页 `/` 的发布页弹窗脚本：自动弹出、关闭、复制域名与焦点管理。

## 职责

1. 页面加载 1.4 秒后自动弹出"永久发布页"模态框（仅首次）
2. 提供"复制域名"功能
3. 完整的无障碍支持（焦点管理、Escape 关闭、aria）
4. 记忆已关闭状态（sessionStorage）

## 完整源码（精简注释版）

```javascript
(function () {
    var STORAGE_KEY = "galnavi-release-modal-dismissed";
    var AUTO_OPEN_DELAY = 1400;
    var modal = document.getElementById("release-modal");
    if (!modal) return;

    var openTriggers = document.querySelectorAll("[data-open-release-modal]");
    var closeTriggers = modal.querySelectorAll("[data-close-release-modal]");
    var copyBtn = modal.querySelector("[data-copy-url]");
    var copyResetTimer = null;
    var lastFocused = null;

    function isOpen() { return modal.classList.contains("active"); }

    function openModal() {
        if (isOpen()) return;
        lastFocused = document.activeElement;       // 记录当前焦点
        modal.classList.add("active");
        modal.setAttribute("aria-hidden", "false");
        document.body.classList.add("modal-open");  // 锁滚动
        var closeBtn = modal.querySelector(".modal-close");
        if (closeBtn) closeBtn.focus();             // 焦点移到关闭键
    }

    function closeModal(remember) {
        if (!isOpen()) return;
        modal.classList.remove("active");
        modal.setAttribute("aria-hidden", "true");
        document.body.classList.remove("modal-open");
        if (remember) {
            try { sessionStorage.setItem(STORAGE_KEY, "1"); } catch (error) {}
        }
        if (lastFocused && typeof lastFocused.focus === "function") {
            lastFocused.focus();                    // 焦点归还
        }
    }

    function shouldAutoOpen() {
        try { return !sessionStorage.getItem(STORAGE_KEY); }
        catch (error) { return true; }
    }

    // 打开触发器
    openTriggers.forEach(function (trigger) {
        trigger.addEventListener("click", function (event) {
            event.preventDefault();
            openModal();
        });
    });
    // 关闭触发器
    closeTriggers.forEach(function (trigger) {
        trigger.addEventListener("click", function () { closeModal(true); });
    });
    // 点遮罩关闭
    modal.addEventListener("click", function (event) {
        if (event.target === modal) { closeModal(true); }
    });
    // Escape 关闭
    document.addEventListener("keydown", function (event) {
        if (event.key === "Escape" && isOpen()) { closeModal(true); }
    });

    // 复制域名按钮
    if (copyBtn) {
        copyBtn.addEventListener("click", function () {
            var text = copyBtn.getAttribute("data-copy-url") || "galnavi.top";
            function markCopied() {
                copyBtn.textContent = "已复制";
                copyBtn.classList.add("copied");
                clearTimeout(copyResetTimer);
                copyResetTimer = setTimeout(function () {
                    copyBtn.textContent = "复制";
                    copyBtn.classList.remove("copied");
                }, 1600);
            }
            if (navigator.clipboard && navigator.clipboard.writeText) {
                navigator.clipboard.writeText(text).then(markCopied).catch(function () { markCopied(); });
            } else {
                markCopied();   // 降级：仍提示已复制
            }
        });
    }

    // 自动弹出
    window.addEventListener("load", function () {
        if (!shouldAutoOpen()) return;
        window.setTimeout(openModal, AUTO_OPEN_DELAY);
    });
})();
```

## 逐段解析

### 1. 自动弹出 + 记忆
```javascript
var AUTO_OPEN_DELAY = 1400;
function shouldAutoOpen() {
    return !sessionStorage.getItem(STORAGE_KEY);
}
window.addEventListener("load", function () {
    if (!shouldAutoOpen()) return;
    window.setTimeout(openModal, AUTO_OPEN_DELAY);
});
```
- 页面 load 后延迟 **1.4 秒**弹出（避免与首屏渲染抢资源）
- 用 **sessionStorage**（非 localStorage）记忆已关闭 → **每次新会话仍会弹一次**
- 这是刻意设计：发布页信息重要，每次访问都提醒，但同会话内不重复打扰
- sessionStorage 不可用时（隐私模式等）默认弹（`return true`）

### 2. 无障碍焦点管理（亮点）
```javascript
function openModal() {
    lastFocused = document.activeElement;   // 1. 记录触发元素
    ...
    closeBtn.focus();                        // 2. 焦点进 modal
}
function closeModal(remember) {
    ...
    lastFocused.focus();                     // 3. 焦点归还
}
```
这是标准的 **modal 焦点陷阱**最佳实践：
- 打开时记录原焦点 → 焦点移入模态框
- 关闭时焦点归还原元素
- 配合 `aria-hidden` 属性切换，屏幕阅读器友好

### 3. 多种关闭方式
- 点关闭按钮（`data-close-release-modal`）
- 点遮罩背景（`event.target === modal`）
- 按 Escape 键
- 三种方式都 `closeModal(true)` 记忆已关闭

### 4. 锁定背景滚动
```javascript
document.body.classList.add("modal-open");    // 开
document.body.classList.remove("modal-open"); // 关
```
通过 body 加类，CSS 中 `.modal-open { overflow: hidden; }` 锁住背景滚动。

### 5. 复制域名
```javascript
var text = copyBtn.getAttribute("data-copy-url") || "galnavi.top";
navigator.clipboard.writeText(text).then(markCopied).catch(function () { markCopied(); });
```
- 优先用现代 `navigator.clipboard` API
- 失败也提示"已复制"（降级，避免报错）
- 1.6 秒后按钮文案从"已复制"恢复"复制"
- 复制内容是 `galnavi.top`（主域名），呼应"永久发布页"用途

### 6. aria 属性
```javascript
modal.setAttribute("aria-hidden", "false"/"true");
```
配合 `aria-hidden` 让屏幕阅读器在模态关闭时跳过该内容。

## 设计亮点

| 设计 | 价值 |
|---|---|
| sessionStorage 记忆 | 每会话提醒一次，不烦人 |
| 1.4s 延迟 | 不抢首屏 |
| 焦点管理 | 无障碍最佳实践 |
| 三种关闭方式 | 用户体验好 |
| clipboard 降级 | 兼容旧环境 |
| `modal-open` 锁滚动 | 防背景滚动穿透 |
