# 帮助页侧栏与锚点脚本

> [!info] 位置
> 帮助页 `/nav/help/` 行175 的 `<script>`。负责侧边栏导航开合、滚动高亮、锚点链接复制。

## 职责

1. 移动端汉堡菜单控制侧栏开合
2. `IntersectionObserver` 实现滚动时侧栏链接高亮
3. 点击锚点图标复制该章节直链到剪贴板

## 完整源码

```javascript
(function() {
    var sidebar = document.getElementById('sidebar');
    var overlay = document.getElementById('overlay');
    var hamburger = document.getElementById('hamburger');

    function openSidebar() {
        sidebar.classList.add('open');
        overlay.classList.add('show');
        document.body.style.overflow = 'hidden';   // 锁滚动
    }
    function closeSidebar() {
        sidebar.classList.remove('open');
        overlay.classList.remove('show');
        document.body.style.overflow = '';
    }
    hamburger.addEventListener('click', function(e) {
        e.stopPropagation();
        sidebar.classList.contains('open') ? closeSidebar() : openSidebar();
    });
    overlay.addEventListener('click', closeSidebar);
    // 点侧栏链接后（移动端）自动收起
    sidebar.querySelectorAll('a').forEach(function(a) {
        a.addEventListener('click', function() {
            if (window.innerWidth <= 768) closeSidebar();
        });
    });

    // ===== 滚动高亮（IntersectionObserver）=====
    var links = sidebar.querySelectorAll('a');
    var sections = [];
    links.forEach(function(a) {
        var id = a.getAttribute('href');
        if (!id || !id.startsWith('#')) return;
        var el = document.querySelector(id);
        if (el) sections.push({ link: a, el: el });
    });
    var activeId = null;
    if (sections.length) {
        var observer = new IntersectionObserver(function(entries) {
            var intersecting = [];
            entries.forEach(function(e) {
                if (e.isIntersecting) intersecting.push(e);
            });
            if (!intersecting.length) return;
            intersecting.sort(function(a, b) {
                return a.boundingClientRect.top - b.boundingClientRect.top;
            });
            var id = '#' + intersecting[0].target.id;
            if (id === activeId) return;
            activeId = id;
            links.forEach(function(a) {
                a.classList.toggle('active', a.getAttribute('href') === id);
            });
        }, { rootMargin: '-8% 0px -72% 0px' });
        sections.forEach(function(s) { observer.observe(s.el); });
    }

    // ===== 锚点复制 toast =====
    var toast = document.getElementById('toast');
    var toastTimer = null;
    document.querySelectorAll('.anchor-icon').forEach(function(el) {
        el.addEventListener('click', function(e) {
            e.preventDefault();
            var url = window.location.origin + window.location.pathname + el.getAttribute('href');
            navigator.clipboard.writeText(url).then(function() {
                showToast();
            }).catch(function() {
                // 降级：execCommand
                var tmp = document.createElement('input');
                tmp.value = url; document.body.appendChild(tmp);
                tmp.select(); document.execCommand('copy'); document.body.removeChild(tmp);
                showToast();
            });
        });
    });
    function showToast() {
        if (toastTimer) clearTimeout(toastTimer);
        toast.classList.add('show');
        toastTimer = setTimeout(function() { toast.classList.remove('show'); }, 1800);
    }
})();
```

## 逐段解析

### 1. 侧栏开合
```javascript
function openSidebar() {
    sidebar.classList.add('open');
    overlay.classList.add('show');
    document.body.style.overflow = 'hidden';
}
```
- 汉堡按钮点击切换开/合
- 遮罩点击关闭
- 移动端（`innerWidth <= 768`）点链接后自动收起侧栏
- 锁滚动用 `body.style.overflow = 'hidden'`（与入口页用 class 不同，但效果一致）

### 2. 滚动高亮（IntersectionObserver）—— 核心
```javascript
var observer = new IntersectionObserver(function(entries) {
    var intersecting = [];
    entries.forEach(function(e) { if (e.isIntersecting) intersecting.push(e); });
    if (!intersecting.length) return;
    // 多个可见时取最靠上的
    intersecting.sort(function(a, b) {
        return a.boundingClientRect.top - b.boundingClientRect.top;
    });
    var id = '#' + intersecting[0].target.id;
    if (id === activeId) return;   // 未变则不操作
    activeId = id;
    links.forEach(function(a) {
        a.classList.toggle('active', a.getAttribute('href') === id);
    });
}, { rootMargin: '-8% 0px -72% 0px' });
```

**工作原理**：
- 收集侧栏所有 `<a href="#xxx">` 与对应章节元素
- 用 IntersectionObserver 监听各章节是否进入视口
- `rootMargin: '-8% 0px -72% 0px'` —— 关键参数：
  - 顶部缩进 8%（忽略刚露出顶部的章节）
  - 底部缩进 72%（只把视口上部约 20% 区域视为"激活区"）
  - 效果：**章节滚到视口上部约 1/5 处时高亮对应链接**，符合阅读直觉
- 多个章节同时可见时，取 `boundingClientRect.top` 最小（最靠上）的
- 用 `activeId` 缓存避免重复操作 DOM

**这是典型的"文档侧栏 TOC 高亮"实现**，比 scroll 事件监听更高效（Observer 异步批量回调，不阻塞滚动）。

### 3. 锚点复制 toast
```javascript
var url = window.location.origin + window.location.pathname + el.getAttribute('href');
navigator.clipboard.writeText(url).then(showToast).catch(function() {
    // 降级 execCommand
    var tmp = document.createElement('input');
    tmp.value = url; document.body.appendChild(tmp);
    tmp.select(); document.execCommand('copy'); document.body.removeChild(tmp);
    showToast();
});
```
- 点击锚点图标（`.anchor-icon`）复制该章节完整 URL
- URL = origin + pathname + hash（如 `https://galnavi.top/nav/help/#guide`）
- 现代 clipboard API 优先，失败降级 `execCommand('copy')`（兼容旧浏览器）
- 复制成功显示 toast 提示 1.8 秒

## 帮助页的章节结构

基于 HTML 中的 `id`，帮助页有以下章节（侧栏 TOC 对应）：

| 锚点 id | 章节内容（推测）|
|---|---|
| `#guide` | 使用指南/入门 |
| `#search` | 搜索说明 |
| `#cards` | 卡片说明 |
| `#detail` | 详情页说明 |
| `#tags` | 标签说明 |
| `#github` | GitHub/贡献说明 |

> 另有 `#hamburger`/`#sidebar`/`#overlay`/`#toast` 是 UI 元素 id，非内容章节。

## 设计亮点

| 设计 | 价值 |
|---|---|
| IntersectionObserver | 比 scroll 高效，不卡顿 |
| rootMargin 精细调参 | 激活时机符合阅读直觉 |
| 多可见取最上 | 滚动过渡时高亮合理 |
| activeId 缓存 | 避免无意义 DOM 操作 |
| clipboard 降级 | 兼容旧环境 |
| toast 反馈 | 复制有明确反馈 |
| 移动端自适应 | 768px 断点收起侧栏 |
